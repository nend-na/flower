        let mode = 'login';
        function toggle() {
            mode = mode === 'login' ? 'signup' : 'login';
            document.getElementById('title').innerText = mode === 'login' ? 'Login' : 'Create Account';
            document.getElementById('extra').style.display = mode === 'login' ? 'none' : 'block';
            document.querySelector('.toggle').innerText = mode === 'login' ? 'New to Reyto? Create Account' : 'Already a member? Login';
        }
        document.getElementById('f').onsubmit = (e) => {
            e.preventDefault();
            localStorage.setItem('isLoggedIn', 'true');
            localStorage.setItem('userEmail', document.getElementById('e').value);
            window.location.href = 'index.html';
        };
