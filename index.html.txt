<script>
    window.onscroll = () => { document.getElementById('nav').classList.toggle('scrolled', window.scrollY > 50); };
    
    const cards = document.querySelectorAll('.card');
    cards.forEach((c, i) => setTimeout(() => c.classList.add('show'), i * 150));
    
    document.getElementById('search').oninput = (e) => {
        const val = e.target.value.toLowerCase();
        cards.forEach(c => c.style.display = c.innerText.toLowerCase().includes(val) ? 'block' : 'none');
    };

    function sel(n, p) {
        localStorage.setItem('selectedFlower', n);
        localStorage.setItem('selectedPrice', p);
        window.location.href = 'booking.html';
    }

    window.onload = () => {
        if(localStorage.getItem('isLoggedIn') === 'true') {
            const user = localStorage.getItem('userEmail').split('@')[0];
            document.getElementById('authBtn').innerText = "LOGOUT";
            document.getElementById('greeting').innerText = "Welcome, " + user.toUpperCase();
            document.getElementById('authBtn').onclick = () => { localStorage.clear(); location.reload(); };
        }
    };
</script>
